Line Graph

            elroi(
              { $el:$('#line').find('.graph'), 
                data: [{series:testSeries, options: {type:'line'}}]
              })
              .draw()

Line Graph With Small Points

            elroi(
              { $el:$('#line').find('.graph'),
                data: [
                  {series:testSeries, 
                    options: {type:'line', showPoints:true, pointStroke:false}}}
                ]
              })
              .draw()

Line Graph With Points

            elroi(
              { $el:$('#line_points').find('.graph'),

              .draw()

Line Graph With Points Animated

            elroi(
              { $el:$('#line_points_animated').find('.graph'),

                        animatePoints:true

                    }

                ]
              })
